/* ---------- PathTrail + HorseshoeTrail ----------
   Builds a smooth Catmull-Rom curve through the node centres,
   an animated turquoise "progress" overlay, and subtle horseshoe
   marks sampled along the real curve between checkpoints.        */
function smoothPath(pts){
  if(pts.length<2) return '';
  let d=`M ${pts[0].x} ${pts[0].y}`;
  for(let i=0;i<pts.length-1;i++){
    const p0=pts[i-1]||pts[i], p1=pts[i], p2=pts[i+1], p3=pts[i+2]||p2;
    const c1x=p1.x+(p2.x-p0.x)/6, c1y=p1.y+(p2.y-p0.y)/6;
    const c2x=p2.x-(p3.x-p1.x)/6, c2y=p2.y-(p3.y-p1.y)/6;
    d+=` C ${c1x.toFixed(1)} ${c1y.toFixed(1)}, ${c2x.toFixed(1)} ${c2y.toFixed(1)}, ${p2.x} ${p2.y}`;
  }
  return d;
}

function PathTrail(svg, points, progressIndex){
  const NS='http://www.w3.org/2000/svg';
  const d = smoothPath(points);

  const base = document.createElementNS(NS,'path');
  base.setAttribute('d', d);
  base.setAttribute('class','trail-line');
  base.setAttribute('stroke-width','15');
  svg.appendChild(base);

  // progress overlay — drawn up to the active checkpoint
  const prog = document.createElementNS(NS,'path');
  prog.setAttribute('d', d);
  prog.setAttribute('class','trail-progress');
  prog.setAttribute('stroke-width','9');
  svg.appendChild(prog);

  const total = base.getTotalLength();
  // length from start to the progress checkpoint
  let target = total;
  if(progressIndex < points.length){
    // approximate: walk the curve to the point nearest the progress node
    const goal = points[progressIndex];
    let best=0, bd=1e9;
    for(let L=0; L<=total; L+=6){
      const p=base.getPointAtLength(L);
      const dist=(p.x-goal.x)**2+(p.y-goal.y)**2;
      if(dist<bd){bd=dist;best=L;}
    }
    target = best;
  }
  prog.style.strokeDasharray = total;
  prog.style.strokeDashoffset = total;          // start fully hidden
  requestAnimationFrame(()=>{
    prog.style.transition='stroke-dashoffset 1.1s cubic-bezier(.16,1,.3,1)';
    prog.style.strokeDashoffset = total - target;
  });

  HorseshoeTrail(svg, base, total, points);
}

function HorseshoeTrail(svg, pathEl, total, points){
  const NS='http://www.w3.org/2000/svg';
  const STEP=44;            // spacing between marks
  const CLEAR=46;           // keep clear of node centres
  let order=0;
  for(let L=STEP; L<total-6; L+=STEP){
    const p = pathEl.getPointAtLength(L);
    // skip marks sitting on top of a checkpoint
    if(points.some(pt=>((pt.x-p.x)**2+(pt.y-p.y)**2) < CLEAR*CLEAR)) continue;
    const a = pathEl.getPointAtLength(Math.max(0,L-4));
    const b = pathEl.getPointAtLength(Math.min(total,L+4));
    const deg = Math.atan2(b.y-a.y, b.x-a.x)*180/Math.PI + 90;
    const g = document.createElementNS(NS,'g');
    g.setAttribute('class','shoe');
    g.setAttribute('transform',`translate(${p.x} ${p.y}) rotate(${deg}) scale(0.62)`);
    g.style.animationDelay = (order*0.05+0.4)+'s';
    g.innerHTML = `
      <path d="M-7 -9 C-12 -3 -12 7 -7 12 M7 -9 C12 -3 12 7 7 12"
            fill="none" stroke="#8A6B3F" stroke-width="3.4" stroke-linecap="round"/>
      <circle cx="-9.5" cy="-7" r="1.1" fill="#8A6B3F"/>
      <circle cx="9.5" cy="-7" r="1.1" fill="#8A6B3F"/>`;
    svg.appendChild(g);
    order++;
  }
}

/* ---------- Terrain : subtle, spacious background accents ----------
   Soft hills, a couple of çinar (plane-tree) silhouettes, a faint
   stone/mud-volcano mound. Deliberately sparse — the screen breathes. */
function Terrain(svg, W, H){
  svg.innerHTML = `
    <defs>
      <linearGradient id="hill" x1="0" y1="0" x2="0" y2="1">
        <stop offset="0" stop-color="#E9F1E2"/><stop offset="1" stop-color="#DCE8D2"/>
      </linearGradient>
    </defs>
    <!-- distant soft hills -->
    <path d="M0 ${H*0.30} Q ${W*0.28} ${H*0.24} ${W*0.55} ${H*0.30} T ${W} ${H*0.31} V0 H0 Z" fill="#F1ECDD" opacity=".5"/>
    <ellipse cx="${W*0.16}" cy="${H*0.46}" rx="120" ry="48" fill="url(#hill)" opacity=".45"/>
    <ellipse cx="${W*0.92}" cy="${H*0.66}" rx="140" ry="52" fill="url(#hill)" opacity=".40"/>
    <ellipse cx="${W*0.10}" cy="${H*0.80}" rx="120" ry="46" fill="url(#hill)" opacity=".40"/>
    <!-- faint mud-volcano mound -->
    <g opacity=".30" fill="#E4D4B6">
      <ellipse cx="${W*0.84}" cy="${H*0.40}" rx="40" ry="14"/>
      <path d="M${W*0.84-22} ${H*0.40} Q ${W*0.84} ${H*0.40-16} ${W*0.84+22} ${H*0.40} Z"/>
    </g>
    <!-- çinar tree silhouettes -->
    ${chinar(W*0.90, H*0.20, 0.9)}
    ${chinar(W*0.08, H*0.62, 0.75)}
    <!-- subtle grass tufts -->
    ${grass(W*0.20, H*0.34)} ${grass(W*0.80, H*0.52)} ${grass(W*0.30, H*0.74)} ${grass(W*0.72, H*0.84)}`;
}
function chinar(x,y,s){
  return `<g transform="translate(${x} ${y}) scale(${s})" opacity=".16" fill="#5E7E4E">
    <rect x="-2.4" y="14" width="4.8" height="26" rx="2" fill="#7A5A38" opacity=".9"/>
    <circle cx="0" cy="2" r="16"/><circle cx="-12" cy="10" r="11"/><circle cx="12" cy="10" r="11"/><circle cx="0" cy="16" r="12"/>
  </g>`;
}
function grass(x,y){
  return `<g transform="translate(${x} ${y})" opacity=".22" stroke="#7FA268" stroke-width="2.4" stroke-linecap="round" fill="none">
    <path d="M0 0 C-2 -8 -3 -10 -3 -12"/><path d="M0 0 C0 -9 0 -12 0 -14"/><path d="M0 0 C2 -8 3 -10 3 -12"/></g>`;
}

/* ---------- QuizModal ---------- */
const modalRoot = document.getElementById('modal-root');
function QuizModal({ kicker, title, img, quiz }){
  let qi = 0;
  modalRoot.innerHTML = '';
  // ASSET: monument thumbnail — hides itself until the real image exists
  const thumb = img ? `<div class="sheet__thumb"><img src="${img}" alt="" onerror="this.closest('.sheet__thumb').style.display='none'"></div>` : '';
  const dots = `<div class="quiz__dots">${quiz.map((_,i)=>`<i class="${i===0?'on':''}"></i>`).join('')}</div>`;
  const sheet = el(`
    <div>
      <div class="modal-scrim"></div>
      <div class="sheet" role="dialog" aria-modal="true" aria-label="${title}">
        <div class="sheet__grip"></div>
        <div class="sheet__head">
          ${thumb}
          <div>
            <div class="sheet__kicker">${kicker}</div>
            <div class="sheet__title">${title}</div>
          </div>
          <button class="sheet__close" aria-label="Bağla">&times;</button>
        </div>
        ${dots}
        <div class="quiz__body"></div>
      </div>
    </div>`);
  modalRoot.appendChild(sheet);

  const body  = sheet.querySelector('.quiz__body');
  const dotsE = sheet.querySelectorAll('.quiz__dots i');

  function close(){
    modalRoot.classList.remove('open');
    modalRoot.setAttribute('aria-hidden','true');
    setTimeout(()=>{ modalRoot.innerHTML=''; }, 320);
  }
  sheet.querySelector('.sheet__close').addEventListener('click', close);
  sheet.querySelector('.modal-scrim').addEventListener('click', close);

  function render(){
    const item = quiz[qi];
    const last = qi === quiz.length-1;
    body.innerHTML = `
      <div class="quiz__q">${item.q}</div>
      <div class="quiz__opts">
        ${item.options.map((o,i)=>`<button class="opt" data-i="${i}">${o}</button>`).join('')}
      </div>
      <div class="quiz__foot">
        <span class="feedback"></span>
        <button class="btn-primary" disabled>${last?'Bitir':'Davam et'}</button>
      </div>`;
    dotsE.forEach((d,i)=>d.classList.toggle('on', i<=qi));

    const opts = body.querySelectorAll('.opt');
    const fb   = body.querySelector('.feedback');
    const next = body.querySelector('.btn-primary');
    let answered = false;

    opts.forEach(opt=>opt.addEventListener('click', ()=>{
      if(answered) return; answered = true;
      const chosen = +opt.dataset.i;
      opts.forEach(o=>o.setAttribute('disabled',''));
      if(chosen === item.answer){
        opt.classList.add('opt--right');
        fb.textContent='Düzdür!'; fb.className='feedback feedback--right';
      }else{
        opt.classList.add('opt--wrong');
        opts[item.answer].classList.add('opt--right');
        fb.textContent='Düzgün cavab işıqlanıb.'; fb.className='feedback feedback--wrong';
      }
      next.removeAttribute('disabled');
    }));

    next.addEventListener('click', ()=>{
      if(last) return close();
      qi++; render();
    });
  }
  render();

  modalRoot.classList.add('open');
  modalRoot.setAttribute('aria-hidden','false');
}

/* ============================================================
   LearningPathScreen — orchestrator + responsive layout
   ============================================================ */
function LearningPathScreen(mount){
  mount.innerHTML='';

  const topbar = TopBar();
  const header = UnitHeaderCard({
    section:1, unit:1, title:'Qədim Azərbaycan',
    onMenu: ()=>QuizModal({
      kicker:'Bölmə 1, Vahid 1', title:'Qədim Azərbaycan',
      quiz:[{ q:'Bu vahid nəyi əhatə edir?',
        options:['Qədim dövr','Müasir dövr','Orta əsrlər','Gələcək'], answer:0 }]
    })
  });

  const stage = el(`
    <main class="stage">
      <svg class="terrain" preserveAspectRatio="none"></svg>
      <svg class="trail"></svg>
      <div class="layer-nodes"></div>
    </main>`);

  const nav = BottomNavigation();

  mount.append(topbar, header, stage, nav);

  const terrainSvg = stage.querySelector('.terrain');
  const trailSvg   = stage.querySelector('.trail');
  const layer      = stage.querySelector('.layer-nodes');

  // index of the checkpoint progress should reach (the active one)
  const progressIdx = Math.max(0, LESSONS.findIndex(l=>l.state==='active'));

  function layout(){
    const W = stage.clientWidth;
    const TOP = 70;                       // top padding before first node
    const GAP = 122;                      // vertical spacing between nodes
    const SWAY = Math.min(96, W*0.24);    // zigzag amplitude
    const cx = W/2;

    const H = TOP + (LESSONS.length-1)*GAP + 150;
    stage.style.minHeight = H+'px';
    trailSvg.setAttribute('viewBox',`0 0 ${W} ${H}`);
    trailSvg.setAttribute('width',W); trailSvg.setAttribute('height',H);

    // node centre points
    const pts = LESSONS.map((l,i)=>({ x: Math.round(cx + l.lane*SWAY), y: TOP + i*GAP }));

    // --- terrain ---
    Terrain(terrainSvg, W, H);

    // --- trail + horseshoes ---
    trailSvg.innerHTML='';
    PathTrail(trailSvg, pts, progressIdx);

    // --- nodes ---
    layer.innerHTML='';
    LESSONS.forEach((l,i)=>{
      const n = LessonNode(l, openLesson);
      n.style.left = pts[i].x+'px';
      n.style.top  = pts[i].y+'px';
      layer.appendChild(n);
    });

    // --- monuments (side challenges) ---
    MONUMENTS.forEach(m=>{
      const li = LESSONS.findIndex(l=>l.id===m.nearLesson);
      const anchor = pts[li] || pts[0];
      const node = MonumentQuizPoint(m, openMonument);
      const x = m.side==='left' ? Math.max(70, W*0.16) : Math.min(W-70, W*0.84);
      node.style.left = x+'px';
      node.style.top  = (anchor.y + 6)+'px';
      layer.appendChild(node);
    });

    // --- mascot guiding just ahead of the active checkpoint ---
    // sit it in the open lane below the active node, on the side the
    // path is NOT swaying toward. A generous vertical offset keeps it
    // clear of both neighbouring nodes and the monument on that row.
    const ap = pts[progressIdx] || pts[0];
    const mascot = MascotGuide();
    const mside = LESSONS[progressIdx].lane > 0 ? 'left' : 'right';
    mascot.style.left = (mside==='left' ? Math.max(82, W*0.21) : Math.min(W-82, W*0.79))+'px';
    mascot.style.top  = (ap.y + 96)+'px';
    layer.appendChild(mascot);
  }

  function openLesson(l){
    QuizModal({
      kicker:'Dərs', title:l.title,
      // placeholder lesson quiz — wire real lesson content here
      quiz:[
        { q:`"${l.title}" dərsinə xoş gəldin. Hazırsan?`,
          options:['Bəli, başlayaq','Bir az sonra'], answer:0 },
        { q:'Azərbaycan hansı qitədədir?',
          options:['Avrasiya','Cənubi Amerika','Afrika','Avstraliya'], answer:0 },
      ]
    });
  }
  function openMonument(m){
    QuizModal({ kicker:'Görməli yer', title:m.label, img:m.img, quiz:m.quiz });
  }

  // responsive: re-layout on resize (debounced)
  let t=null;
  window.addEventListener('resize', ()=>{ clearTimeout(t); t=setTimeout(layout,150); });
  layout();
}

LearningPathScreen(document.getElementById('app'));
